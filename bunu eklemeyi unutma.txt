def should_trust_prediction(rsi, bb_pctb, vol_ratio):
    if rsi < 0.40 or rsi > 0.60:
        return False, "RSI ekstrem bölgede"
    if bb_pctb < 0.10 or bb_pctb > 0.90:
        return False, "Bollinger breakout"
    if vol_ratio > 2.0:
        return False, "Hacim patlaması"
    return True, "Güvenilir tahmin"
